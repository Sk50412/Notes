
## variable syntax
data type   variable name = 0 (or) null (or) value


## object syntax
class_name object_variable_name = new class_name (parameters)


## method syntax
method_modifier   return_type method_name (parameters) { // method body}


## if-else syntax
if (expression)
{
	(true_statement)
}
else if (expression)
{
	(else_if statements)
}
else
{
	else statements
}



## Switch syntax
switch (expression)
{
	case (1) - statements
				 break
	case (2) - statements
				 break
	case (n) - statements
				break
	default - statements
				  break
}




## advanced switch case
switch (expression)
{
	case 1 -> statements;
	case 2 -> statemetns;
	default -> staements;
}


## expression casting
(data_type) (var_name) = (data_type_to_convert) (expression)



## casting syntax
desired_type variable;



## class syntax
class_modifier class class_name 
<extends superclass_name> (or) <implements interface_1 , interface_2 , ... > { /// class methods}
